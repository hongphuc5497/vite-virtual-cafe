Install Playwright as a devDependency, create `playwright.config.js` with webServer config pointing at port 3000, write shared test constants and page-object helpers. No tests yet — this is the foundation that 03-02 and 03-03 build on.

<task number="4" goal="Create tests/helpers/mainPage.js">
<action>
Create `tests/helpers/mainPage.js` with page-object helpers for the main `/` route:

```js
import { expect } from '@playwright/test';
import { SELECTORS, TIMEOUTS, ROUTES } from '../constants';

export class MainPage {
  constructor(page) {
    this.page = page;
  }

  async goto() {
    await this.page.goto(ROUTES.HOME);
    await this.page.waitForLoadState('networkidle');
  }

  async startTimer(preset = '25') {
    const selector = SELECTORS[`TIMER_PRESET_${preset}`];
    await this.page.click(selector);
    await this.page.click(SELECTORS.TIMER_START);
  }

  async getTimerText() {
    const el = this.page.locator(SELECTORS.TIMER_DISPLAY);
    return el.textContent();
  }

  async waitForCelebration() {
    const overlay = this.page.locator(SELECTORS.CELEBRATION_OVERLAY);
    await overlay.waitFor({ state: 'visible', timeout: TIMEOUTS.CELEBRATION });
    return overlay;
  }

  async getCelebrationHeading() {
    const el = this.page.locator(SELECTORS.CELEBRATION_HEADING);
    return el.textContent();
  }

  async selectVibe(vibe) {
    await this.page.click(SELECTORS.VIBE_OPTION(vibe));
  }

  async selectScene(sceneName) {
    await this.page.click(`${SELECTORS.SCENE_SELECTOR} [data-testid=scene-${sceneName}]`);
  }

  async getLocalStorage(key) {
    return this.page.evaluate((k) => localStorage.getItem(k), key);
  }

  async isMixerVisible() {
    return this.page.locator(SELECTORS.MIXER_PANEL).isVisible();
  }
}
```

  async goto() {

    return this.page.locator(SELECTORS.RELAX_AMBIENT_PLAYER);
  }

  async isAmbientPlayerVisible() {
    return this.page.locator(SELECTORS.RELAX_AMBIENT_PLAYER).isVisible();
  }

  async timerDisplayExists() {
    return this.page.locator(SELECTORS.TIMER_DISPLAY).isVisible();
  }

  async getLocalStorage(key) {
    return this.page.evaluate((k) => localStorage.getItem(k), key);
  }

  async setLocalStorage(key, value) {
    await this.page.evaluate(({k, v}) => localStorage.setItem(k, v), {k: key, v: value});
  }
}
